4  Câmera

Por padrão, a câmera é definida com uma relação de pixel 16:9, dessa forma as dimensões são, aproximadamente: 8.0 unidades de altura e 14.222 unidades de largura. As dimensões da câmera são independentes da resolução de renderização (480p, 720p, etc.) estipuladas pelo usuário, portanto, não há necessidade de se preocupar com isso.

Código 
from manim import *


class CameraDimension(Scene):
    def construct(self):
        def frame_annotation(framew, frameh, pixelw, pixelh):
            marca_unid_x = DoubleArrow(framew * LEFT / 2, framew * RIGHT / 2, buff=0).to_edge(DOWN)
            marca_unid_y = DoubleArrow(frameh * UP / 2, frameh * DOWN / 2, buff=0).to_edge(LEFT)

            marca_px_x = DoubleArrow(framew * LEFT / 2, framew * RIGHT / 2, buff=0).to_edge(UP)
            marca_px_y = DoubleArrow(frameh * UP / 2, frameh * DOWN / 2, buff=0).to_edge(RIGHT)

            txt_unid_x = Text(str(framew)[:5] + " unidades", font_size=30).next_to(marca_unid_x, UP)
            txt_unid_y = Text(str(frameh) + " unidades", font_size=30).next_to(marca_unid_y, RIGHT)
            
            txt_px_x = Text(str(pixelw) + " pixels", font_size=30).next_to(marca_px_x, DOWN)
            txt_px_y = Text(str(pixelh) + " pixels", font_size=30).next_to(marca_px_y, LEFT)

            unidades = Group(marca_unid_x,marca_unid_y,txt_unid_x,txt_unid_y).set_color(PURE_GREEN)
            pixels = Group(marca_px_x,marca_px_y,txt_px_x,txt_px_y).set_color(PURE_RED)

            return (unidades,pixels)
        
        dimensao = frame_annotation(config.frame_width, config.frame_height, 
        config.pixel_width, config.pixel_height)

        self.add(*dimensao)
Dica

As resoluções comuns para 16:9 estão listadas em 16:9 aspect ratio.

            txt_unid_y = Text(str(frameh) + " unidades", font_size=30).next_to(marca_unid_y, RIGHT)
            
            txt_px_x = Text(str(pixelw) + " pixels", font_size=30).next_to(marca_px_x, DOWN)

            pixels = Group(marca_px_x,marca_px_y,txt_px_x,txt_px_y).set_color(PURE_RED)

            return (unidades, pixels)
        
        config.frame_height = 8 # Valor considerado para a altura
        
        dimensao = frame_annotation(config.frame_width, config.frame_height, 
        config.pixel_width, config.pixel_height)

        self.add(*dimensao, Circle(fill_opacity=1, radius=0.3, color=PURE_BLUE).to_edge(RIGHT))

4.2 Cenas da câmera

Em todos os exemplos apresentados até agora, utilizamos um tipo de cena chamado Scene, que é o tipo básico para realizar animações. Agora, vamos explorar os tipos de cenas mais utilizados no manim.

4.2.1 moving_camera_scene

Uma cena com configurações e propriedades especiais que a tornam adequada para casos em que a câmera deve ser movimentada.

Código Figura 4.11 
from manim import *
from scipy.stats import chi2


class ChiSquare(MovingCameraScene):
    def construct(self):
        self.camera.frame.save_state()
        
        colors = [PINK, PURE_RED, PURE_GREEN, YELLOW]

        # Textos
        main = Tex("Distribuição $\chi^2$ com $k$ graus de liberdade", tex_template=TexFontTemplates.comic_sans)
        
        equation = MathTex("f(x)=\\frac{x^{k/2-1}e^{-x/2}}{2^{k/2}\\Gamma(k/2)}, \quad x\\ge0").scale(0.8)

        framebox = SurroundingRectangle(equation, buff = .1, color="#49A88F")

        ########################################################################
        ### Gráfico qui-quadrado
        ########################################################################

        ax = Axes(x_range=[0, 20], y_range=[0, 0.2], tips = False)
        graph = ax.plot(lambda x: chi2.pdf(x, df=4), color=BLUE, x_range=[0, 20])
        
        self.play(FadeIn(main), FadeIn(txt[2:], shift=UP))
        self.wait()
        self.play(main.animate.to_edge(UP), run_time=2)
        self.play(FadeIn(ax, graph), equation.animate.move_to(RIGHT*2.5), run_time=2)
        self.play(self.camera.frame.animate.scale(0.5).move_to(RIGHT*2.5), 
                  Create(framebox.move_to(RIGHT*2.5)), run_time=2)
        self.wait(2)
        self.play(Restore(self.camera.frame), FadeOut(framebox), run_time=2)
        self.wait(2)
        self.play(FadeOut(graph), FadeOut(equation))
                
        for i in range(1,5):
            graph2 = ax.plot(lambda x: chi2.pdf(x, df=2*i), color=colors[1*i-1], x_range=[0, 20])
            gl = Group(Tex("$k= $").shift(RIGHT*2.), Integer(2*i).shift(RIGHT*2.7)).move_to(UR*3)
            self.play(FadeIn(Group(graph2, gl.shift(DOWN*i).set_color(colors[1*i-1]))))
            self.wait()

        self.wait(2)

4.2.2 scene

Tela básica para animações.

Código Figura 4.12 
from manim import *
import itertools
import random


class PontosAleatorios(Scene):
    config = {
        "PontosTotal" : 9,
        }
    def construct(self):
       
       colors = itertools.cycle([PURE_RED, PURE_GREEN, PURE_BLUE, YELLOW, PURPLE, PINK, ORANGE, TEAL, GOLD])
       itemsTotal = PontosAleatorios.config["PontosTotal"]

       d = VGroup()
       for i in range(itemsTotal):
            d = d.add(Dot(color = next(colors)).copy().scale(10))

       d.arrange_in_grid(cols=3, buff=0.1)
       self.play(Write(d))
       
       [self.play(CyclicReplace(d[i], d[j])) for i,j in random.sample(list(itertools.product(range(itemsTotal), repeat = 2)), 10)]
            
       [self.play(Unwrite(d[i])) for i in random.sample(range(itemsTotal),itemsTotal)]

4.2.3 three_d_scene

Uma cena adequada para renderizar objetos e animações tridimensionais.

Código Figura 4.15 
from manim import *


class Quadricas(ThreeDScene):
    def construct(self):

        eixos = ThreeDAxes()

        def param_cilindro(u, v):
            x, y = np.cos(TAU * v), np.sin(TAU * v)
            z = 2 * (1 - u)
            return np.array([x, y, z])
        
        cilindro = Surface(
            param_cilindro,
            resolution=(6, 32),
            v_range=[-PI, PI],
            u_range=[0, PI/2],
            checkerboard_colors=[BLUE, GREEN_E])

        def param_paraboloide(u, v):
            x, y = np.cos(v)*u, np.sin(v)*u
            z = u**2
            return np.array([x, y, z])
        
        paraboloide = Surface(
            param_paraboloide,
            resolution=(15, 32),
            v_range=[-PI, PI],
            u_range=[0, PI/2],
            checkerboard_colors=[BLUE, TEAL])

        def param_hyperboloide(u, v):
            x, y = u, v
            z = u**2-v**2
            return np.array([x, y, z])
        
        hyperboloide = Surface(
            param_hyperboloide,
            resolution=(15, 32),
            v_range=[-2, 2],
            u_range=[-2, 2],
            checkerboard_colors=[YELLOW_A, YELLOW_E]).scale(0.5)
        
        def param_cone(u, v):
            x, y = u*np.cos(v), u*np.sin(v)
            z = u
            return np.array([x, y, z])
        
        cone = Surface(
            param_cone,
            resolution=(15, 32),
            v_range=[0, TAU],
            u_range=[-2, 2],
            checkerboard_colors=[GREEN_A, GREEN_B]).scale(1)
        
        def param_hip_uma_face(u, v):
            x, y = np.cosh(u)*np.cos(v), np.cosh(u)*np.sin(v)
            z = np.sinh(u)
            return np.array([x, y, z])
        
        hip_uma_face = Surface(
            param_hip_uma_face,
            resolution=(15, 32),
            v_range=[0, TAU],
            u_range=[-2, 2],
            checkerboard_colors=[PINK, LIGHT_PINK]).scale(0.5)
        
        def param_elipsoide(u, v):
            x, y = 1*np.cos(u)*np.cos(v), 2*np.cos(u)*np.sin(v)
            z = 0.5*np.sin(u)
            return np.array([x, y, z])
        
        elipsoide = Surface(
            param_elipsoide,
            resolution=(15, 32),
            v_range=[0, TAU],
            u_range=[-PI/2, PI/2],
            checkerboard_colors=[PURPLE, PURPLE_A]).scale(1.5)
        
        def param_esfera(u, v):
            x, y = 1.5*np.cos(u)*np.cos(v), 1.5*np.cos(u)*np.sin(v)
            z = 1.5*np.sin(u)
            return np.array([x, y, z])
        
        esfera = Surface(
            param_esfera,
            resolution=(15, 32),
            v_range=[0, TAU],
            u_range=[-PI/2, PI/2],
            checkerboard_colors=[ORANGE, RED]).scale(1)
        
        self.set_camera_orientation(phi=75 * DEGREES)
        self.begin_ambient_camera_rotation(rate=0.2)


        self.add(eixos,cone)
        self.wait(2)
        self.play(ReplacementTransform(cone, hip_uma_face))
        self.wait()
        self.play(ReplacementTransform(hip_uma_face, cilindro))
        self.wait()
        self.play(ReplacementTransform(cilindro, hyperboloide))
        self.wait()
        self.play(ReplacementTransform(hyperboloide, paraboloide))
        self.wait()
        self.play(ReplacementTransform(paraboloide, elipsoide))
        self.wait()
        self.play(ReplacementTransform(elipsoide, esfera))
        self.wait(2)
        self.play(FadeOut(eixos, esfera))
